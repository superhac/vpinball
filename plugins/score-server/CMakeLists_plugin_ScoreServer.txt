# plugin CMake file to be included in main project CMakelists

# --- ScoreServer plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(SCORESERVER_PLUGIN_SOURCES
   plugins/score-server/score-server.cpp
)

if(BUILD_SHARED)
   add_library(ScoreServerPlugin MODULE ${SCORESERVER_PLUGIN_SOURCES})

   target_include_directories(ScoreServerPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/score-server
   )

   if(PluginPlatform STREQUAL "windows")
      add_compile_options(
         $<$<CONFIG:RELEASE>:/Ob2>
         $<$<CONFIG:RELEASE>:/O2>
         $<$<CONFIG:RELEASE>:/Oi>
         $<$<CONFIG:RELEASE>:/arch:SSE2>
         $<$<CONFIG:RELEASE>:/fp:fast>
         $<$<CONFIG:RELEASE>:/fp:except->
         $<$<CONFIG:RELEASE>:/Ot>
         $<$<CONFIG:RELEASE>:/GF>
         $<$<CONFIG:RELEASE>:/GS->
         $<$<CONFIG:RELEASE>:/Gy>
         $<$<CONFIG:RELEASE>:/GR>
         $<$<CONFIG:RELEASE>:/Oy>
         $<$<CONFIG:RELEASE>:/GT>
         $<$<CONFIG:RELEASE>:/GL>
      )
      set_target_properties(ScoreServerPlugin PROPERTIES
         MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
      )
      target_compile_options(ScoreServerPlugin PUBLIC
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
      )
      target_link_options(ScoreServerPlugin PUBLIC
         $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
         $<$<CONFIG:RELEASE>:/OPT:REF>
         $<$<CONFIG:RELEASE>:/OPT:ICF>
         $<$<CONFIG:RELEASE>:/LTCG>
      )
      if (PluginArch STREQUAL "x64")
         set_target_properties(ScoreServerPlugin PROPERTIES
            OUTPUT_NAME "plugin-score-server64"
         )
      else()
         set_target_properties(ScoreServerPlugin PROPERTIES
            OUTPUT_NAME "plugin-score-server"
         )
      endif()
   else()
      set_target_properties(ScoreServerPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-score-server"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(ScoreServerPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
   endif()

   add_custom_command(TARGET ScoreServerPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/score-server/plugin.cfg" "${PLUGINS_DIR}/score-server/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy_directory "${CMAKE_SOURCE_DIR}/plugins/score-server/nvram-maps" "${PLUGINS_DIR}/score-server/nvram-maps"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:ScoreServerPlugin>" "${PLUGINS_DIR}/score-server"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:ScoreServerPlugin>"
   )
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(ScoreServerPlugin STATIC ${SCORESERVER_PLUGIN_SOURCES})

   target_include_directories(ScoreServerPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/score-server
   )

   set_target_properties(ScoreServerPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-score-server"
   )
endif()
